Add these to your package.json "scripts":

  "start": "expo start",
  "start:tunnel": "expo start --tunnel",
  "start:lan": "expo start --lan",
  "android": "expo start --android",
  "ios": "expo start --ios",
  "build:android:preview": "eas build --profile preview --platform android",
  "build:android:prod": "eas build --profile production --platform android",
  "build:ios:prod": "eas build --profile production --platform ios",
  "submit:play": "eas submit -p android",
  "submit:appstore": "eas submit -p ios"
